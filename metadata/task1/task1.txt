FROM t_user_relate r,t_user u

FROM t_user_relate r,t_user u
WHERE fansid=1994626454 AND u.uid=r.uid

#导出微博列表
SELECT 	tid, uid,retid,created_at,TEXT,source_name,reposts_count,comments_count,originalPic,create_date
